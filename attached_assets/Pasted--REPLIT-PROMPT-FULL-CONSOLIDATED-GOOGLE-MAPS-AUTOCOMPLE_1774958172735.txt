**CONTEXT — READ FIRST:**

Phes has two locations:
- **Oak Lawn** — serves all Oak Lawn zones
- **Schaumburg** — serves all Schaumburg zones

Every zip code in the system must route to exactly one location. No zip can route to both. No zip can fall through to the wrong location. Routing is determined at the zip level, enforced at the database level, and validated at the widget level. This is non-negotiable.

**PART 3 — BACKEND: ZIP ROUTING API ENDPOINT**

Create `GET /api/public/service-zones/check` — no auth required, used by the booking widget:

```javascript
// Query params: zip, companyId or companySlug
app.get('/api/public/service-zones/check', async (req, res) => {
  const { zip, companySlug } = req.query;
  
  if (!zip || !companySlug) {
    return res.status(400).json({ error: 'zip and companySlug required' });
  }

  // Get company
  const company = await db.query(
    'SELECT id FROM companies WHERE slug = $1', 
    [companySlug]
  );
  
  if (!company.rows.length) {
    return res.status(404).json({ inZone: false, location: null });
  }

  const companyId = company.rows[0].id;

  // Check all zones for this company
  // zip_codes is stored as an array — check if zip exists in any zone
  const result = await db.query(`
    SELECT name, location, color
    FROM service_zones
    WHERE company_id = $1
    AND $2 = ANY(zip_codes)
    LIMIT 1
  `, [companyId, zip]);

  if (result.rows.length === 0) {
    return res.json({ 
      inZone: false, 
      location: null, 
      zoneName: null 
    });
  }

  return res.json({ 
    inZone: true, 
    location: result.rows[0].location,  // 'oak_lawn' or 'schaumburg'
    zoneName: result.rows[0].name,
    color: result.rows[0].color
  });
});
```

This is the single source of truth for zip routing. The widget never uses hardcoded zip lists — it always queries this endpoint.

**ZIP CONFLICT VALIDATION (critical):**

```javascript
const validateNoZipConflicts = async (companyId, zips, excludeZoneId = null) => {
  const query = excludeZoneId
    ? `SELECT name, location, zip_codes 
       FROM service_zones 
       WHERE company_id = $1 AND id != $2`
    : `SELECT name, location, zip_codes 
       FROM service_zones 
       WHERE company_id = $1`;

  const params = excludeZoneId ? [companyId, excludeZoneId] : [companyId];
  const zones = await db.query(query, params);

  const conflicts = [];
  for (const zone of zones.rows) {
    for (const zip of zips) {
      if (zone.zip_codes.includes(zip)) {
        conflicts.push({ 
          zip, 
          existingZone: zone.name,
          existingLocation: zone.location 
        });
      }
    }
  }
  return conflicts;
};
```

If conflicts are found, return 400 with the conflict details so the office knows exactly which zone already has that zip.

**PART 5 — GOOGLE MAPS AUTOCOMPLETE IN BOOKING WIDGET**

In the booking widget at `/book/phes-cleaning`, wire up Google Maps Places Autocomplete on the Service Address field in Step 1.

**Load Maps script:**

```javascript
// In the booking widget component
useEffect(() => {
  const apiKey = import.meta.env.VITE_GOOGLE_MAPS_API_KEY;
  
  if (!apiKey) {
    console.error('[Maps] VITE_GOOGLE_MAPS_API_KEY not set');
    return;
  }

  if (window.google?.maps?.places) {
    initAutocomplete();
    return;
  }

  window.__mapsCallback = () => initAutocomplete();
  
  const script = document.createElement('script');
  script.src = `https://maps.googleapis.com/maps/api/js?key=${apiKey}&libraries=places&callback=__mapsCallback&loading=async`;
  script.async = true;
  script.defer = true;
  script.onerror = () => console.error('[Maps] Script failed to load');
  document.head.appendChild(script);

  return () => {
    delete window.__mapsCallback;
  };
}, []);
```

**Initialize autocomplete:**

```javascript
const initAutocomplete = () => {
  const input = addressInputRef.current;
  if (!input) return;

  const autocomplete = new window.google.maps.places.Autocomplete(input, {
    componentRestrictions: { country: 'us' },
    fields: ['address_components', 'formatted_address', 'geometry'],
    types: ['address']
  });

  autocomplete.addListener('place_changed', async () => {
    const place = autocomplete.getPlace();
    if (!place.address_components) return;

    const get = (type) =>
      place.address_components.find(c => c.types.includes(type))?.long_name || '';
    const shortGet = (type) =>
      place.address_components.find(c => c.types.includes(type))?.short_name || '';

    const zip = get('postal_code');
    const addressData = {
      formatted: place.formatted_address,
      street: `${get('street_number')} ${get('route')}`.trim(),
      city: get('locality'),
      state: shortGet('administrative_area_level_1'),
      zip,
      lat: place.geometry.location.lat(),
      lng: place.geometry.location.lng(),
      verified: true
    };

    setServiceAddress(addressData.formatted);
    setAddressComponents(addressData);
    setZipCode(zip); // auto-fill zip field
    setAddressVerified(true);

    // Check zone via API — never use hardcoded lists
    if (zip) {
      await checkZone(zip);
    }
  });
};
```

**Zone check function:**

```javascript
const checkZone = async (zip) => {
  try {
    setZoneStatus('checking');
    const res = await fetch(
      `/api/public/service-zones/check?zip=${zip}&companySlug=phes-cleaning`
    );
    const data = await res.json();
    
    if (data.inZone) {
      setZoneStatus('in_zone');
      setBookingLocation(data.location); // 'oak_lawn' or 'schaumburg'
    } else {
      setZoneStatus('out_of_zone');
      setBookingLocation(null);
    }
  } catch {
    setZoneStatus(null);
  }
};
```

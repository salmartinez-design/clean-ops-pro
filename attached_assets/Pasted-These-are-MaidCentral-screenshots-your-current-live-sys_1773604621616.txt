DATABASE:
  ALTER TABLE users ADD COLUMN IF NOT EXISTS 
    invite_token TEXT,
    invite_sent_at TIMESTAMP,
    invite_accepted_at TIMESTAMP,
    onboarding_complete BOOLEAN DEFAULT false;

API ROUTES:

POST /api/employees/invite
  Body: { email, first_name, last_name, role }
  1. Create user record with status 'invited'
  2. Generate secure invite token (crypto.randomBytes(32))
  3. Store token + sent_at in users table
  4. Send invite email via Resend:
     Subject: "You've been invited to join [Company Name]"
     Body: 
       "Hi [first_name],
        [Company Name] has invited you to join 
        CleanOps Pro — your work management portal.
        
        Click below to create your account:
        [Accept Invitation button]
        
        This link expires in 7 days."
     Link: https://[domain]/accept-invite?token=[token]
  5. Return { success: true, invite_sent_to: email }

GET /api/invite/:token
  Validates token exists, not expired (7 days),
  not already accepted.
  Returns { valid: true, employee_name, company_name }
  
POST /api/invite/:token/accept
  Body: { password }
  1. Validate token
  2. Set user password (bcrypt hash)
  3. Set invite_accepted_at = NOW()
  4. Set onboarding_complete = true
  5. Return JWT token for immediate login

DATABASE ADDITIONS

CREATE TABLE IF NOT EXISTS employee_availability (
  id          SERIAL PRIMARY KEY,
  company_id  INTEGER NOT NULL REFERENCES companies(id),
  user_id     INTEGER NOT NULL REFERENCES users(id),
  day_of_week INTEGER NOT NULL, -- 0=Sun, 1=Mon...6=Sat
  start_time  TIME,
  end_time    TIME,
  is_available BOOLEAN DEFAULT true
);

CREATE TABLE IF NOT EXISTS attendance_records (
  id          SERIAL PRIMARY KEY,
  company_id  INTEGER NOT NULL REFERENCES companies(id),
  user_id     INTEGER NOT NULL REFERENCES users(id),
  date        DATE NOT NULL,
  status      VARCHAR(20) NOT NULL 
              CHECK (status IN (
                'worked','pto','sick','time_off',
                'excused','unexcused','holiday'
              )),
  hours       DECIMAL(4,2),
  notes       TEXT,
  created_by  INTEGER REFERENCES users(id),
  created_at  TIMESTAMP DEFAULT NOW()
);

CREATE TABLE IF NOT EXISTS time_off_requests (
  id          SERIAL PRIMARY KEY,
  company_id  INTEGER NOT NULL REFERENCES companies(id),
  user_id     INTEGER NOT NULL REFERENCES users(id),
  type        VARCHAR(20) NOT NULL,
  start_date  DATE NOT NULL,
  end_date    DATE NOT NULL,
  hours       DECIMAL(4,2),
  status      VARCHAR(10) DEFAULT 'pending'
              CHECK (status IN ('pending','approved','denied')),
  notes       TEXT,
  reviewed_by INTEGER REFERENCES users(id),
  reviewed_at TIMESTAMP,
  created_at  TIMESTAMP DEFAULT NOW()
);

CREATE TABLE IF NOT EXISTS scorecard_entries (
  id          SERIAL PRIMARY KEY,
  company_id  INTEGER NOT NULL REFERENCES companies(id),
  user_id     INTEGER NOT NULL REFERENCES users(id),
  job_id      INTEGER REFERENCES jobs(id),
  score       DECIMAL(2,1) NOT NULL CHECK (score BETWEEN 0 AND 4),
  category    VARCHAR(50),
  notes       TEXT,
  created_by  INTEGER REFERENCES users(id),
  created_at  TIMESTAMP DEFAULT NOW()
);

CREATE TABLE IF NOT EXISTS client_ratings (
  id          SERIAL PRIMARY KEY,
  company_id  INTEGER NOT NULL REFERENCES companies(id),
  client_id   INTEGER NOT NULL REFERENCES clients(id),
  job_id      INTEGER NOT NULL REFERENCES jobs(id),
  score       INTEGER NOT NULL CHECK (score BETWEEN 1 AND 4),
  comment     TEXT,
  created_at  TIMESTAMP DEFAULT NOW()
);

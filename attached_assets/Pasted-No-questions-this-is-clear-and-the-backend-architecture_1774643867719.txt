**PART 1 — DATABASE SCHEMA**

Create two new tables:

```sql
CREATE TABLE addon_bundles (
  id SERIAL PRIMARY KEY,
  company_id INTEGER NOT NULL REFERENCES companies(id) ON DELETE CASCADE,
  name VARCHAR(100) NOT NULL,
  description TEXT,
  discount_type VARCHAR(20) NOT NULL DEFAULT 'flat_per_item',
  -- flat_per_item: $X off each item in bundle
  -- flat_total: $X off the total bundle price
  -- percentage: X% off each item in bundle
  discount_value NUMERIC(10,2) NOT NULL DEFAULT 0,
  active BOOLEAN NOT NULL DEFAULT true,
  valid_from DATE,
  valid_until DATE,
  created_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE addon_bundle_items (
  id SERIAL PRIMARY KEY,
  bundle_id INTEGER NOT NULL REFERENCES addon_bundles(id) ON DELETE CASCADE,
  addon_key VARCHAR(100) NOT NULL
  -- addon_key matches the key used in pricing_addons table
);
```
